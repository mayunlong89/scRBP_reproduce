```R
# Read data
df <- read.csv("./Supplementary_Figure_S2B-C/spearman_correlation_heatmap.csv")
# Remove the first column (RBP names)
rank_matrix <- df[ , -1]
# Convert all columns to numeric (skip RBP name column)
rank_numeric <- as.data.frame(lapply(rank_matrix, function(x) as.numeric(as.character(x))))
# Now compute Spearman correlation (automatically handles NA)
cor_matrix<-cor(rank_numeric, method = "spearman", use = "pairwise.complete.obs")
# Visualization
library(pheatmap)
pheatmap(cor_matrix, main = "Spearman Correlation Between Ranking Methods")
# Generate color scheme
my_colors <- colorRampPalette(c("white","#F6DFC0","#FEA087","#B2182B"))(100)
# Plot
pheatmap(cor_matrix,
         color = my_colors,
         cluster_rows = F,
         cluster_cols = F,
         display_numbers = TRUE,        # Display values
         number_format = "%.2f",        # Decimal places to keep
         main = "Spearman Correlation Between Ranking Methods")
```
